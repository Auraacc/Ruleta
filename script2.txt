const sectors = document.querySelectorAll(".sector");
const pin = document.querySelector(".pin");
const resultContainer = document.getElementById("resultContainer");
let rotation = 0;

function spin() {
  const randomAngle = Math.floor(Math.random() * 360);
  const fullTurns = 5; // Ajusta el número de vueltas completas (5 por defecto)

  rotation = fullTurns * 360 + randomAngle;
  const rotationAngle = `rotate(${rotation}deg)`;

  document.querySelector(".wheel").style.transition = "transform 5s ease-out";
  document.querySelector(".wheel").style.transform = rotationAngle;
  
  setTimeout(showResult, 6000); // Mostrar resultado después de 6 segundos (ajusta según necesites)
}

function showResult() {
  const sectorAngle = 360 / sectors.length;
  const sectorIndex = Math.floor((rotation % 360) / sectorAngle);
  const selectedValue = sectors[sectorIndex].getAttribute("data-sector");
  resultContainer.textContent = `¡Has ganado un descuento de: ${selectedValue}!`;
}
